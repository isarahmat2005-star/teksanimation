from manim import * class TextMotionApp(Scene): def construct(self): # 1. TERIMA INPUT DARI PENGGUNA (Bisa diganti variabel dinamis) teks_baris_1 = "HIDUP" teks_baris_2 = "JOKOWI" # 2. SETUP GAYA (STYLE) # Membuat teks dengan font tebal dan warna sesuai video contoh t1 = Text(teks_baris_1, font="Arial", weight=BOLD, font_size=96, color=TEAL_C) t2 = Text(teks_baris_2, font="Arial", weight=BOLD, font_size=96, color=WHITE) # Garis bawah dekoratif (seperti di video) garis = Line(start=LEFT, end=RIGHT, color=GREEN).width(4).next_to(t2, DOWN) # Mengatur posisi (Grouping) grup_teks = VGroup(t1, t2).arrange(DOWN, buff=0.2) # 3. BACKGROUND (Opsional: Meniru efek gelap/partikel) self.camera.background_color = "#0f1712" # Hijau tua sangat gelap # 4. DEFINISI MOTION (ANIMASI) # Animasi 1: Fade In + Geser Sedikit (Slide) self.play( FadeIn(t1, shift=UP), FadeIn(t2, shift=UP), Create(garis), run_time=1.5 ) # Animasi 2: Sedikit "Zoom" (Scale) untuk efek dramatis self.play( grup_teks.animate.scale(1.1), garis.animate.width(5), run_time=3 ) # Animasi 3: Fade Out self.play(FadeOut(grup_teks), FadeOut(garis))
